## Quick Example

Here's how easy it is to get started with Joyfill:

<CodeGroup>

```swift iOS
import Joyfill

// Initialize
Joyfill.configure(accessToken: "your_token")

// Load form
let formView = JoyfillFormView()
let config = JoyfillFormConfig(
    mode: .fill,
    templateId: "your_template_id"
)
formView.loadForm(config: config)
```

```kotlin Android
import io.joyfill.Joyfill

// Initialize
Joyfill.configure("your_token")

// Load form
val formView = JoyfillFormView(context)
val config = JoyfillFormConfig(
    mode = FormMode.FILL,
    templateId = "your_template_id"
)
formView.loadForm(config)
```

```javascript React Native
import { JoyfillForm } from '@joyfill/react-native';

// Initialize
Joyfill.configure('your_token');

// Load form
<JoyfillForm
  mode="fill"
  templateId="your_template_id"
  onSubmit={(data) => console.log(data)}
/>
```

```javascript Web
import { Joyfill } from '@joyfill/web';

// Initialize
Joyfill.configure('your_token');

// Load form
const form = new Joyfill.Form({
  mode: 'fill',
  templateId: 'your_template_id',
  container: '#form-container'
});
form.load();
```

```bash REST API
# Create a document from template
curl -X POST https://api-joy.joyfill.io/v1/documents \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "New Document",
    "group": "group_id",
    "template": "template_id"
  }'
```
